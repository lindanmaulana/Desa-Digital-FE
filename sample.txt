import { SidebarContent, SidebarGroup, SidebarGroupContent, SidebarGroupLabel, SidebarMenu, SidebarMenuButton, SidebarMenuItem } from '@/components/ui/sidebar';
import { Link } from 'react-router';

interface PathUrl {
	id: number;
	title: string;
	url: string;
	icon: string;
	iconActive: string;
}
const items: PathUrl[] = [
	{
		id: 1,
		title: 'Dashboard',
		url: '/dashboard',
		icon: 'chart-square-secondary-green.svg',
		iconActive: 'chart-square-dark-green.svg',
	},
];

interface ContentAppSidebarProps {
	pathname: string
}

export const ContentAppSidebar = ({pathname}: ContentAppSidebarProps) => {
	return (
		<SidebarContent>
			<SidebarGroup>
				<SidebarGroupLabel className="text-sm text-village-secondary font-normal">Main Menu</SidebarGroupLabel>
				<SidebarGroupContent>
					<SidebarMenu>
						{items.map((item) => {
							const isActive = item.url === pathname

							return (
							<SidebarMenuItem key={item.id}>
								<SidebarMenuButton asChild>
									<Link to={item.url} className={`${isActive && "bg-village-forshadow"} w-full h-full p-4 text-[16px] font-medium rounded-xl`}>
										<img src={`/images/icons/${isActive ? item.iconActive : item.icon}`} className='size-6 object-contain' />

									</Link>
								</SidebarMenuButton>
							</SidebarMenuItem>
						)})}
					</SidebarMenu>
				</SidebarGroupContent>
			</SidebarGroup>
		</SidebarContent>
	);
};
